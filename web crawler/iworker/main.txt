                from.html(html);
                from.submit();
            }

    $(function(){
        var $container = $('.change_skin'),
            $body = $('body');
        $container.on('click', '.caret', function(){
            var lists = $container.find('.theme_lists')[0],
                $t = $(this);
            if($t.is('.left')){
                lists.scrollLeft -= 140;
            }else{
                lists.scrollLeft += 140;
            }
//            if(lists.scrollLeft + lists)
        }).on('click.toggle.change_skin', '.theme_lists>li', function(){
            var $t = $(this),
                theme = $t.data('toggle_class'),
                $active = $container.find('.active'),
                remove_cls = $active.data('toggle_class') + '_theme',
                add_cls = theme + '_theme';

            iwk_cfg.theme = theme;
            $body.removeClass(remove_cls).addClass(add_cls);
            $active.removeClass('active');
            $t.addClass('active');

            //传递至后台更改redis的theme
            var url = iwk_cfg.siteurl + 'inside/dash/ajax/index/action:switch_dash_theme',
                data = {theme: theme};
            iwk_utils.post(url, data, function(res){
                if(res.ret != ERP_CONSTANT.RET_SUCCESS)
                    iwk_utils.alert_error(res.msg);
            });
        }).on('click.close.change_skin', '.close', function(){
            $container.toggleClass('open');
        });
        iwk_workflow_relate_create2.init();
        iwk_relation_workflow2.init();
        iwk_relation_workflow_picker.init();
    });
</script>

<script type="text/javascript">
    $(function(){
        var $accounting_arrange_modal = $("#accounting_arrange_modal");

        // 为“整理”按钮绑定点击事件
        $accounting_arrange_modal.off("click", "button.arrange").on("click", "button.arrange", function(){
            var $this = $(this),
                $accounting_arrange_result_modal = $("#accounting_arrange_result_modal"),
                gl_year = $accounting_arrange_modal.find(".arrange_period").data("gl_year"),
                gl_period = $accounting_arrange_modal.find(".arrange_period").data("gl_period"),
                is_include_audited = $accounting_arrange_modal.find(".is_include_audited [name='is_include_audited']:checked").val() || "0",
                sort_type = $accounting_arrange_modal.find(".sort_type [name='sort_type']:checked").val(),
                post_url = "erp/api/erp_voucher/action:fix_voucher_number",
                post_data = {
                    gl_year: gl_year,
                    gl_period: gl_period,
                    is_include_audited: is_include_audited,
                    sort_type: sort_type
                };

            // 将数据发送后台
            $this.button("loading");
            iwk_utils.post(post_url, post_data, function(res){
                $this.button("reset");

                if(res.ret == 0){
                    iwk_utils.alert_success("整理成功");

                    // 关闭整理操作的模态框
                    $accounting_arrange_modal.modal("hide");

                    // 展示整理结果的模态框
                    $accounting_arrange_result_modal.modal("show");
                    var option = {
                        cls: 'erp_datagrid',
                        loadMsg: '数据加载中...',
                        empty_msg: '处理中，请刷新查看整理结果...',
                        fitColumns: true,
                        columns: [[
                            {field: 'gl_year_period', title: '会计期间', width: 140},
                            {field: 'voucher_word_name', title: '凭证字', width: 140},
                            {field: 'before_voucher_number', title: '整理前凭证号', width: 140},
                            {field: 'after_voucher_number', title: '整理后凭证号', width: 140}
                        ]],
                        onLoadSuccess: function(res){
                            if(res.ret == 0){
                                var $t = $(this);
                                if(res.is_finished == 1)
                                    $t.datagrid('options').empty_msg = '暂无数据';
                            }
                        },
                        pagination: false,
                        url: iwk_cfg.siteurl + 'erp/api/erp_voucher/action:get_voucher_fix_result',
                        queryParams: {
                            gl_year: gl_year,
                            gl_period: gl_period
                        }
                    };

                    // 生成datagrid
                    $accounting_arrange_result_modal.find(".accounting_arrange_result_table").datagrid(option);
                }else{
                    iwk_utils.alert_error(res.msg);
                }
            });
        });
    });
</script>
